impl Solution {
    pub fn two_sum(nums: Vec<i32>, target: i32) -> Vec<i32> {
        let mut result_map: HashMap<i32, i32> = HashMap::new();

        for (idx, val) in nums.iter().enumerate() {
            match result_map.get(&(target - *val)) {
                Some(&idx2) => return vec![idx2, idx as i32],
                None => result_map.insert(*val, idx as i32),
            };
        }

        return vec![];
    }
}
