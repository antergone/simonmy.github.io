#!/usr/bin/python3
# -*- coding: utf-8 -*-

import base64
import random
import time

import requests
from google.protobuf import descriptor as _descriptor
from google.protobuf import descriptor_pool as _descriptor_pool
from google.protobuf import symbol_database as _symbol_database
from google.protobuf.internal import builder as _builder

# @@protoc_insertion_point(imports)

_sym_db = _symbol_database.Default()

DESCRIPTOR = _descriptor_pool.Default().AddSerializedFile(
    b'\n\nyang.proto\",\n\x05\x43hess\x12\x12\n\nchessIndex\x18\x01 \x01(\x05\x12\x0f\n\x07timeTag\x18\x02 \x01(\x05\"_\n\rMatchPlayInfo\x12\x10\n\x08gameType\x18\x01 \x01(\x05\x12\r\n\x05mapId\x18\x02 \x01(\x05\x12\x0f\n\x07mapSeed\x18\x03 \x01(\x05\x12\x1c\n\x0cstepInfoList\x18\x04 \x03(\x0b\x32\x06.Chessb\x06proto3')

_builder.BuildMessageAndEnumDescriptors(DESCRIPTOR, globals())
_builder.BuildTopDescriptorsAndMessages(DESCRIPTOR, 'yang_pb2', globals())
if _descriptor._USE_C_DESCRIPTORS == False:
    DESCRIPTOR._options = None
    _CHESS._serialized_start = 14
    _CHESS._serialized_end = 58
    _MATCHPLAYINFO._serialized_start = 60
    _MATCHPLAYINFO._serialized_end = 155
# @@protoc_insertion_point(module_scope)


describe = '''
usage: python|python3 main.py

description: 该程序用于羊了个羊快速通关, 可刷新通关次数
author: simon.my
website: www.simonmy.com
update: 2022-09-24 15:45:00
'''

# 配置项
token = '此处替换成自己的TOKEN'
execute_times = 999

headers = {
    'Accept': '*/*',
    'Accept-Encoding': 'gzip,compress,br,deflate',
    'Connection': 'keep-alive',
    'content-type': 'application/json',
    # Android使用下面的 UA和 Refer
    # 'Referer': 'https://servicewechat.com/wx141bfb9b73c970a9/27/page-frame.html',
    # 'User-Agent': 'Mozilla/5.0 (Linux; Android 12; M2012K11C Build/SKQ1.211006.001; wv) AppleWebKit/537.36 (KHTML, like Gecko) Version/4.0 Chrome/86.0.4240.99 XWEB/4313 MMWEBSDK/20220805 Mobile Safari/537.36 MMWEBID/4629 MicroMessenger/8.0.27.2220(0x28001B37) WeChat/arm64 Weixin NetType/WIFI Language/zh_CN ABI/arm64 MiniProgramEnv/android',
    'Referer': 'https://servicewechat.com/wx141bfb9b73c970a9/27/page-frame.html',
    'User-Agent': 'Mozilla/5.0 (iPhone; CPU iPhone OS 14_6 like Mac OS X) AppleWebKit/605.1.15 (KHTML, like Gecko) Mobile/15E148 MicroMessenger/8.0.27(0x18001b37) NetType/WIFI Language/zh_CN',

    't': token
}

get_map_url = 'https://cat-match.easygame2021.com/sheep/v1/game/map_info_ex?matchType=3'
report_url = 'https://cat-match.easygame2021.com/sheep/v1/game/game_over_ex?'
get_personal_info_url = 'https://cat-match.easygame2021.com/sheep/v1/game/personal_info?'
get_map_data_url = 'https://cat-match-static.easygame2021.com/maps/{}.txt'


def getTime():
    return time.strftime("%Y-%m-%d-%H_%M_%S", time.localtime())


# 获取当前获胜数量
def get_win_count():
    if token == '':
        print(getTime(), "请配置token")
        exit(1)
    try:
        result = requests.get(get_personal_info_url, headers=headers).json()
        if result['err_code'] != 0:
            print(getTime(), result['err_msg'])
            print(getTime(), "token过期或配置不正确")
            exit(1)
        else:
            win_count = result['data']['win_count']
            print(getTime(), result['data']['nick_name'], ": 当前通关次数为", win_count)
            return win_count
    except Exception:
        print(getTime(), "请求失败, 请检查token过期或配置不正确")
        exit(1)


def fetch_map_info(map_md5):
    url = get_map_data_url.format(map_md5)
    r = requests.get(url)
    return r.json()['levelData']


def encrypt_match_play_info(level_data):
    info = MatchPlayInfo()
    info.gameType = 3
    for idx in range(len(sum(level_data.values(), []))):  # 生成操作序列
        chess = Chess()
        chess.chessIndex = idx
        chess.timeTag = random.randint(100, 600)
        info.stepInfoList.append(chess)
    return base64.b64encode(info.SerializeToString()).decode('utf-8')


def execute(finished_times):
    
    level_data = {}
    map_md5 = ''
    for i in range(0, execute_times):
        r = requests.get(get_map_url, headers=headers)
        
        current_map_md5 = r.json()['data']['map_md5'][1]

        # 检查地图是否更新，重新下载新的地图数据
        if current_map_md5 != map_md5:
            print(getTime(), "服务器更新地图, 重新下载新的地图数据, 今日地图md5:", current_map_md5)
            map_md5 = current_map_md5
            level_data = fetch_map_info(map_md5)
        else:
            print(getTime(), "服务器未更新地图, 使用上一次地图数据, 今日地图md5:", current_map_md5)

        match_play_info = encrypt_match_play_info(level_data)
        # 休眠60s, 否则无法更新记录成功
        time.sleep(60)
        result = requests.post(report_url, headers=headers,
                               json={'rank_score': 1, 'rank_state': 1, 'rank_time': 20, 'rank_role': 1, 'skin': 1,
                                     'MatchPlayInfo': match_play_info})
        print(getTime(), '上传结果', result.text)
        # 检查是否更新成功        
        info = requests.get(get_personal_info_url, headers=headers).json()
        newWinCount = info['data']['win_count']
        if newWinCount == finished_times:
            print(getTime(), '更新失败, 可能被羊了个羊小程序拉黑')
        else:
            print(getTime(), '更新成功, 通关次数', finished_times, '->', newWinCount)
            finished_times = newWinCount


# 主函数
def main():
    print(describe)
    win_count = get_win_count()
    print(getTime(), '刷新通关次数(每次需要1分钟):', execute_times)
    execute(win_count)


main()
